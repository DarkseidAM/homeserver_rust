3use super::{CURRENT_SCHEMA_VERSION, HistoryRepo};
4use crate::history_repo::aggregation;
5use std::path::Path;

8use sqlx::sqlite::{SqliteConnectOptions, SqlitePoolOptions};

10impl HistoryRepo {
11    pub async fn connect(path: &str, retention_days: u32) -> anyhow::Result<Self> {
12        if let Some(parent) = Path::new(path).parent() {
13            std::fs::create_dir_all(parent)?;
14        }
15        let opts = SqliteConnectOptions::from_str(&format!("sqlite:{}", path))?
16            .create_if_missing(true)
17            .journal_mode(sqlx::sqlite::SqliteJournalMode::Wal)
18            .busy_timeout(std::time::Duration::from_secs(5))
19            .synchronous(sqlx::sqlite::SqliteSynchronous::Normal);
20        let pool = SqlitePoolOptions::new().connect_with(opts).await?;
21        let retention_ms = (retention_days as i64) * 24 * 60 * 60 * 1000;
22        Ok(Self { pool, retention_ms })
23    }
24
25    async fn drop_history_user_tables(
26        tx: &mut sqlx::Transaction<'_, sqlx::Sqlite>,
27    ) -> anyhow::Result<()> {
28        sqlx::query("DROP TABLE IF EXISTS system_history")
29            .execute(&mut **tx)
30            .await?;
31        sqlx::query("DROP TABLE IF EXISTS system_history_aggregated")
32            .execute(&mut **tx)
33            .await?;
34        sqlx::query("DROP TABLE IF EXISTS system_info")
35            .execute(&mut **tx)
36            .await?;
37        Ok(())
38    }
39
40    async fn ensure_schema_version(&self) -> anyhow::Result<()> {
41        sqlx::query(
42            "CREATE TABLE IF NOT EXISTS schema_version (key TEXT PRIMARY KEY, value INTEGER NOT NULL)",
43        )
44        .execute(&self.pool)
45        .await?;
46
47        let row: Option<i64> =
48            sqlx::query_scalar("SELECT value FROM schema_version WHERE key = 'schema'")
49                .fetch_optional(&self.pool)
50                .await?;
51
52        match row {
53            None => {
54                let legacy_tables: i64 = sqlx::query_scalar(
55                    r#"SELECT COUNT(*) FROM sqlite_master
56                       WHERE type = 'table'
57                         AND name IN ('system_history', 'system_info', 'system_history_aggregated')"#,
58                )
59                .fetch_one(&self.pool)
60                .await?;
61
62                if legacy_tables > 0 {
63                    tracing::warn!(
64                        "schema version row missing but history tables present; purging history"
65                    );
66                    let mut tx = self.pool.begin().await?;
67                    Self::drop_history_user_tables(&mut tx).await?;
68                    sqlx::query(
69                        r#"INSERT INTO schema_version (key, value) VALUES ('schema', $1)
70                           ON CONFLICT(key) DO UPDATE SET value = excluded.value"#,
71                    )
72                    .bind(i64::from(CURRENT_SCHEMA_VERSION))
73                    .execute(&mut *tx)
74                    .await?;
75                    tx.commit().await?;
76                } else {
77                    sqlx::query(
78                        r#"INSERT INTO schema_version (key, value) VALUES ('schema', $1)
79                           ON CONFLICT(key) DO NOTHING"#,
80                    )
81                    .bind(i64::from(CURRENT_SCHEMA_VERSION))
82                    .execute(&self.pool)
83                    .await?;
84                }
85            }
86            Some(v) if v == i64::from(CURRENT_SCHEMA_VERSION) => {}
87            Some(found) => {
88                tracing::warn!(
89                    "schema version mismatch: found {}, expected {}; purging history",
90                    found,
91                    CURRENT_SCHEMA_VERSION
92                );
93                let mut tx = self.pool.begin().await?;
94                Self::drop_history_user_tables(&mut tx).await?;
95                sqlx::query("UPDATE schema_version SET value = $1 WHERE key = 'schema'")
96                    .bind(i64::from(CURRENT_SCHEMA_VERSION))
97                    .execute(&mut *tx)
98                    .await?;
99                tx.commit().await?;
100            }
101        }
102
103        Ok(())
104    }
105
106    pub async fn init(&self) -> anyhow::Result<()> {
107        self.ensure_schema_version().await?;
108
109        sqlx::query(
110            r#"
111            CREATE TABLE IF NOT EXISTS system_history (
112                id INTEGER PRIMARY KEY AUTOINCREMENT,
113                created_at INTEGER NOT NULL,
114                cpu_load REAL NOT NULL,
115                memory_used INTEGER NOT NULL,
116                container_data BLOB NOT NULL,
117                storage_data BLOB NOT NULL,
118                network_data BLOB NOT NULL,
119                system_data BLOB NOT NULL
120            )
121            "#,
122        )
123        .execute(&self.pool)
124        .await?;
125
126        sqlx::query(
127            "CREATE INDEX IF NOT EXISTS idx_history_created_at ON system_history(created_at)",
128        )
129        .execute(&self.pool)
130        .await?;
131
132        sqlx::query(
133            "CREATE TABLE IF NOT EXISTS system_info (id INTEGER PRIMARY KEY CHECK (id = 1), data BLOB NOT NULL)",
134        )
135        .execute(&self.pool)
136        .await?;
137
138        aggregation::init_aggregated_table(&self.pool).await?;
139
140        Ok(())
141    }
