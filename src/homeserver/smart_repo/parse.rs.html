3use crate::models::SmartHealth;

6/// Parse `smartctl --scan -j` output into a list of device paths.
7pub fn parse_scan_devices(json: &str) -> Vec<String> {
8    let Ok(v) = serde_json::from_str::<Value>(json) else {
9        return Vec::new();
10    };
11    v["devices"]
12        .as_array()
13        .map(|arr| {
14            arr.iter()
15                .filter_map(|d| d["name"].as_str().map(str::to_string))
16                .collect()
17        })
18        .unwrap_or_default()
19}

21/// Parse `smartctl -a -j <device>` output into a [`SmartHealth`]. Returns None only if the JSON
22/// is unparseable; missing individual fields degrade to None/false/empty.
23pub fn parse_smartctl_json(json: &str, device: &str) -> Option<SmartHealth> {
24    let v: Value = serde_json::from_str(json).ok()?;
25    Some(SmartHealth {
26        device: device.to_string(),
27        model: v["model_name"].as_str().unwrap_or("").to_string(),
28        health_passed: v["smart_status"]["passed"].as_bool().unwrap_or(false),
29        temperature_c: v["temperature"]["current"].as_i64(),
30        power_on_hours: v["power_on_time"]["hours"].as_u64(),
31        reallocated_sectors: ata_attr_raw(&v, 5),
32        wear_level_percent: wear_level(&v),
33    })
34}

36/// Raw value of an ATA SMART attribute by id (from `ata_smart_attributes.table`).
37fn ata_attr_raw(v: &Value, id: u64) -> Option<u64> {
38    v["ata_smart_attributes"]["table"]
39        .as_array()?
40        .iter()
41        .find(|a| a["id"].as_u64() == Some(id))
42        .and_then(|a| a["raw"]["value"].as_u64())
43}

45/// SSD/NVMe life used as a percentage. Prefer NVMe `percentage_used`; else derive from an ATA
46/// wear-leveling attribute's normalized value (which counts down from 100 = full life remaining).
47fn wear_level(v: &Value) -> Option<u8> {
48    if let Some(used) = v["nvme_smart_health_information_log"]["percentage_used"].as_u64() {
49        return Some(used.min(100) as u8);
50    }
51    let table = v["ata_smart_attributes"]["table"].as_array()?;
52    table
53        .iter()
54        .find(|a| matches!(a["id"].as_u64(), Some(177) | Some(231) | Some(233)))
55        .and_then(|a| a["value"].as_u64())
56        .map(|life_remaining| (100u64.saturating_sub(life_remaining)).min(100) as u8)
57}
