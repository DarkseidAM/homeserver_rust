26impl SysinfoRepo {
27    pub fn new() -> Self {
28        let mut sys = System::new_all();
29        sys.refresh_all();
30        let disks = Disks::new_with_refreshed_list();
31        let networks = Networks::new_with_refreshed_list();
32        Self {
33            sys: Arc::new(std::sync::Mutex::new(sys)),
34            disks: Arc::new(std::sync::Mutex::new(disks)),
35            networks: Arc::new(std::sync::Mutex::new(networks)),
36            last_network: Arc::new(std::sync::Mutex::new(None)),
37            last_cpu_refresh: Arc::new(std::sync::Mutex::new(None)),
38        }
39    }

41    #[instrument(skip(self), fields(repo = "sysinfo", operation = "get_cpu_stats"))]
42    pub async fn get_cpu_stats(&self) -> anyhow::Result<CpuStats> {
43        let sys = self.sys.clone();
44        let last_cpu_refresh = self.last_cpu_refresh.clone();
45        tokio::task::spawn_blocking(move || {
46            let mut sys = sys
47                .lock()
48                .map_err(|e| anyhow::anyhow!("sysinfo lock poisoned: {}", e))?;
49
50            let now = Instant::now();
51            let usage = if let Ok(mut guard) = last_cpu_refresh.lock() {
52                if let Some((prev_ts, prev_usage)) = *guard {
53                    let dt = now.duration_since(prev_ts);
54                    if dt >= sysinfo::MINIMUM_CPU_UPDATE_INTERVAL {
55                        // Enough time has passed, refresh and get new usage
56                        sys.refresh_cpu_all();
57                        let new_usage = sys.global_cpu_usage() as f64;
58                        *guard = Some((now, new_usage));
59                        new_usage
60                    } else {
61                        // Not enough time has passed, return cached usage without blocking
62                        prev_usage
63                    }
64                } else {
65                    // First call: refresh to establish baseline
66                    sys.refresh_cpu_all();
67                    *guard = Some((now, 0.0));
68                    0.0
69                }
70            } else {
71                // Lock failed, refresh and return 0.0
72                sys.refresh_cpu_all();
73                0.0
74            };
75
76            let physical = System::physical_core_count().unwrap_or(0) as u32;
77            let logical = sys.cpus().len() as u32;
78            let name = linux::read_cpu_model_linux()
79                .or_else(|| {
80                    sys.cpus()
81                        .first()
82                        .map(|c| c.name().to_string())
83                        .filter(|s| !s.is_empty() && s != "cpu0")
84                })
85                .unwrap_or_else(|| "Unknown".into());
86
87            let core_usages: Vec<f64> = sys
88                .cpus()
89                .iter()
90                .map(|c| (c.cpu_usage() as f64).clamp(0.0, 100.0))
91                .collect();
92            let temperature = linux::read_cpu_temperature_linux().unwrap_or(0.0);
93
94            Ok(CpuStats {
95                model: name,
96                physical_cores: physical,
97                logical_cores: logical,
98                usage_percent: usage.clamp(0.0, 100.0),
99                temperature,
100                core_usages,
101            })
102        })
103        .await
104        .map_err(|e| anyhow::anyhow!("sysinfo task join: {}", e))?
105    }

107    #[instrument(skip(self), fields(repo = "sysinfo", operation = "get_ram_stats"))]
108    pub async fn get_ram_stats(&self) -> anyhow::Result<RamStats> {
109        let sys = self.sys.clone();
110        tokio::task::spawn_blocking(move || {
111            let mut sys = sys
112                .lock()
113                .map_err(|e| anyhow::anyhow!("sysinfo lock poisoned: {}", e))?;
114            sys.refresh_memory();
115
116            let total = sys.total_memory();
117            let available = sys.available_memory();
118            let used = total.saturating_sub(available);
119            let usage_percent = if total > 0 {
120                (used as f64 / total as f64) * 100.0
121            } else {
122                0.0
123            };
124
125            Ok(RamStats {
126                total,
127                used,
128                available,
129                usage_percent,
130                swap_total: sys.total_swap(),
131                swap_used: sys.used_swap(),
132                swap_free: sys.free_swap(),
133            })
134        })
135        .await
136        .map_err(|e| anyhow::anyhow!("sysinfo task join: {}", e))?
137    }
