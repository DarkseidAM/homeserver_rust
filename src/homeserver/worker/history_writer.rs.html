12/// Spawns the background task that receives snapshots from the worker and flushes to the DB.
13/// Flushes when buffer len >= flush_rate, or every flush_interval_secs, or when channel closes.
14/// When the worker drops its sender, this task flushes remaining and exits.
15pub fn spawn_history_writer(
16    mut write_rx: mpsc::Receiver<FullSystemSnapshot>,
17    history_repo: Arc<HistoryRepo>,
18    system_info: Arc<SystemInfo>,
19    config: HistoryWriterConfig,
20    snapshots_saved_total: Arc<AtomicU64>,
21) -> tokio::task::JoinHandle<()> {
22    let flush_interval = Duration::from_secs(config.flush_interval_secs);
23    let persist_gpu = config.persist_gpu;
24    let persist_smart = config.persist_smart;
25    tokio::spawn(async move {
26        let mut buffer: Vec<FullSystemSnapshot> = Vec::new();
27        let mut flush_tick = interval(flush_interval);
28        flush_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
29
30        loop {
31            tokio::select! {
32                result = write_rx.recv() => {
33                    match result {
34                        Some(mut snapshot) => {
35                            if !persist_gpu {
36                                snapshot.gpus.clear();
37                            }
38                            if !persist_smart {
39                                snapshot.smart.clear();
40                            }
41                            buffer.push(snapshot);
42                            if buffer.len() >= config.flush_rate as usize
43                                && let Err(e) = flush_buffer(&history_repo, &system_info, &mut buffer, &snapshots_saved_total).await
44                            {
45                                tracing::warn!(error = %e, "history writer: save_snapshots failed");
46                            }
47                        }
48                        None => break,
49                    }
50                }
51                _ = flush_tick.tick() => {
52                    if let Err(e) = flush_buffer(&history_repo, &system_info, &mut buffer, &snapshots_saved_total).await {
53                        tracing::warn!(error = %e, "history writer: save_snapshots failed");
54                    }
55                }
56            }
57        }
58        if let Err(e) = flush_buffer(
59            &history_repo,
60            &system_info,
61            &mut buffer,
62            &snapshots_saved_total,
63        )
64        .await
65        {
66            tracing::warn!(error = %e, "history writer: final flush failed");
67        }
68        tracing::debug!("History writer shutting down");
69    })
70}
71
72async fn flush_buffer(
73    history_repo: &HistoryRepo,
74    system_info: &SystemInfo,
75    buffer: &mut Vec<FullSystemSnapshot>,
76    snapshots_saved_total: &AtomicU64,
77) -> anyhow::Result<()> {
78    if buffer.is_empty() {
79        return Ok(());
80    }
81    let n = buffer.len();
82    history_repo.save_snapshots(buffer, system_info).await?;
83    snapshots_saved_total.fetch_add(n as u64, std::sync::atomic::Ordering::Relaxed);
84    buffer.clear();
85    tracing::debug!(
86        operation = "save_snapshots",
87        snapshots_count = n,
88        "Snapshots saved"
89    );
90    Ok(())
91}
