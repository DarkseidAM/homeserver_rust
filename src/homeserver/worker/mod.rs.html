19/// Rate limit for "no receivers" warning (avoid logging every second when no one is on /ws/system)
20const NO_RECEIVERS_WARN_INTERVAL: Duration = Duration::from_secs(60);
21
22/// Channel capacity for snapshot writer (backpressure if writer falls behind).
23pub fn writer_channel_capacity(flush_rate: u64) -> usize {
24    (flush_rate as usize * 2).max(32)
25}

44/// Worker timing and logging config.
45/// Stats logging and pruning use real-time intervals, independent of sample_interval_ms.
46pub struct WorkerConfig {
47    pub sample_interval_ms: u64,
48    /// How often to log app stats (real seconds).
49    pub stats_log_interval_secs: u64,
50    /// How often to prune old data (real seconds).
51    pub prune_interval_secs: u64,
52    /// Collect GPU metrics each tick.
53    pub collect_gpu: bool,
54    /// Collect SMART disk health (slow background poll).
55    pub collect_smart: bool,
56    /// How often to refresh SMART data (real seconds).
57    pub smart_poll_interval_secs: u64,
58}
59
60/// Writer config: batching for the dedicated history writer task.
61pub struct HistoryWriterConfig {
62    pub flush_rate: u64,
63    pub flush_interval_secs: u64,
64    /// When false, GPU data is dropped before persisting (live WS still includes it).
65    pub persist_gpu: bool,
66    /// When false, SMART data is dropped before persisting (live WS still includes it).
67    pub persist_smart: bool,
68}

70pub fn spawn(deps: WorkerDeps, config: WorkerConfig) -> tokio::task::JoinHandle<()> {
71    let WorkerDeps {
72        sysinfo_repo,
73        system_info: _,
74        docker_repo,
75        gpu_repo,
76        smart_repo,
77        history_repo,
78        tx,
79        write_tx,
80        ws_system_connections,
81        snapshots_saved_total,
82        mut alert_engine,
83        notifier,
84        mut shutdown_rx,
85    } = deps;
86    let WorkerConfig {
87        sample_interval_ms,
88        stats_log_interval_secs,
89        prune_interval_secs,
90        collect_gpu,
91        collect_smart,
92        smart_poll_interval_secs,
93    } = config;
94
95    let stats_log_interval = Duration::from_secs(stats_log_interval_secs);
96    let prune_interval = Duration::from_secs(prune_interval_secs);
97
98    tokio::spawn(async move {
99        let mut tick = interval(Duration::from_millis(sample_interval_ms));
100        tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
101        let mut stats_log_tick = interval(stats_log_interval);
102        stats_log_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
103        let mut prune_tick = interval(prune_interval);
104        prune_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
105        let mut smart_tick = interval(Duration::from_secs(smart_poll_interval_secs.max(1)));
106        smart_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
107
108        let mut snapshots_pruned_total: u64 = 0;
109        let mut last_no_receivers_warn: Option<Instant> = None;
110
111        let worker_span = tracing::span!(tracing::Level::DEBUG, "worker", sample_interval_ms);
112        let _guard = worker_span.enter();
113
114        loop {
115            tokio::select! {
116                _ = tick.tick() => {
117            let timestamp = std::time::SystemTime::now()
118                .duration_since(std::time::UNIX_EPOCH)
119                .map(|d| d.as_millis() as u64)
120                .unwrap_or_else(|e| {
121                    tracing::warn!(
122                        error = %e,
123                        operation = "get_timestamp",
124                        "system time error"
125                    );
126                    0
127                });
128
129            // Degrade gracefully: a single failing collector falls back to defaults for that
130            // metric rather than dropping the whole tick (which would lose the healthy metrics too).
131            let cpu = sysinfo_repo.get_cpu_stats().await.unwrap_or_else(|e| {
132                tracing::warn!(error = %e, operation = "get_cpu_stats", "CPU stats failed; using defaults");
133                Default::default()
134            });
135            let ram = sysinfo_repo.get_ram_stats().await.unwrap_or_else(|e| {
136                tracing::warn!(error = %e, operation = "get_ram_stats", "RAM stats failed; using defaults");
137                Default::default()
138            });
139            let containers = docker_repo.list_running_and_refresh_stats().await;
140            let storage = sysinfo_repo.get_storage_stats().await.unwrap_or_else(|e| {
141                tracing::warn!(error = %e, operation = "get_storage_stats", "storage stats failed; using defaults");
142                Default::default()
143            });
144            let network = sysinfo_repo.get_network_stats().await.unwrap_or_else(|e| {
145                tracing::warn!(error = %e, operation = "get_network_stats", "network stats failed; using defaults");
146                Default::default()
147            });
148            let system = sysinfo_repo.get_system_stats().await.unwrap_or_else(|e| {
149                tracing::warn!(error = %e, operation = "get_system_stats", "system stats failed; using defaults");
150                Default::default()
151            });
152            // GPU collection does blocking sysfs reads / NVML ioctls — offload to the blocking
153            // pool so it never stalls the async executor (and other tasks like WS connections).
154            let gpus = if collect_gpu {
155                let gpu_repo = gpu_repo.clone();
156                tokio::task::spawn_blocking(move || gpu_repo.collect())
157                    .await
158                    .unwrap_or_default()
159            } else {
160                Vec::new()
161            };
162            // SMART is refreshed on its own slow cadence (smart_tick); read the cached value here.
163            let smart = smart_repo.current();
164
165            let snapshot = FullSystemSnapshot {
166                timestamp,
167                cpu,
168                ram,
169                containers,
170                storage,
171                network,
172                system,
173                gpus,
174                smart,
175            };
176
177            // Evaluate alert rules and dispatch any fire/resolve events (webhook POST is detached).
178            if !alert_engine.is_empty() {
179                for ev in alert_engine.evaluate(&snapshot, std::time::Instant::now()) {
180                    let notifier = notifier.clone();
181                    tokio::spawn(async move { notifier.notify(&ev).await });
182                }
183            }
184
185            // Only clone for the broadcast when someone is actually listening.
186            if tx.receiver_count() > 0 {
187                let _ = tx.send(snapshot.clone());
188            } else {
189                let should_warn = last_no_receivers_warn
190                    .is_none_or(|t| t.elapsed() >= NO_RECEIVERS_WARN_INTERVAL);
191                if should_warn {
192                    tracing::debug!(
193                        operation = "broadcast_snapshot",
194                        "No active WebSocket clients; skipping broadcast"
195                    );
196                    last_no_receivers_warn = Some(Instant::now());
197                }
198            }
199            if write_tx.send(snapshot).await.is_err() {
200                tracing::debug!("History writer channel closed");
201            }
202                }
203                _ = &mut shutdown_rx => {
204                    tracing::debug!("Worker shutting down");
205                    break;
206                }
207                _ = stats_log_tick.tick() => {
208                    tracing::info!(
209                        ws_system_clients =
210                            ws_system_connections.load(std::sync::atomic::Ordering::Relaxed),
211                        snapshots_saved_total = snapshots_saved_total.load(std::sync::atomic::Ordering::Relaxed),
212                        snapshots_pruned_total = snapshots_pruned_total,
213                        "app stats"
214                    );
215                }
216                _ = smart_tick.tick() => {
217                    // smartctl is slow/blocking; refresh in a detached task so the loop stays responsive.
218                    if collect_smart {
219                        let repo = smart_repo.clone();
220                        tokio::spawn(async move { repo.refresh().await });
221                    }
222                }
223                _ = prune_tick.tick() => {
224                    if let Err(e) = history_repo.prune_old_data().await {
225                        tracing::warn!(
226                            error = %e,
227                            operation = "prune_old_data",
228                            "Failed to prune old data"
229                        );
230                    } else {
231                        tracing::debug!(operation = "prune_old_data", "Old data pruned successfully");
232                        snapshots_pruned_total += 1;
233                    }
234                }
235            }
236        }
237    })
