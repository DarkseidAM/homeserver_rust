16/// Rate limit for "no receivers" warning (avoid logging every second when no one is on /ws/system)
17const NO_RECEIVERS_WARN_INTERVAL: Duration = Duration::from_secs(60);
18
19/// Channel capacity for snapshot writer (backpressure if writer falls behind).
20pub fn writer_channel_capacity(flush_rate: u64) -> usize {
21    (flush_rate as usize * 2).max(32)
22}

24/// Repos, channels, and shutdown for the worker.
25pub struct WorkerDeps {
26    pub sysinfo_repo: Arc<SysinfoRepo>,
27    pub system_info: Arc<SystemInfo>,
28    pub docker_repo: Arc<DockerRepo>,
29    pub history_repo: Arc<HistoryRepo>,
30    pub tx: broadcast::Sender<FullSystemSnapshot>,
31    pub write_tx: mpsc::Sender<FullSystemSnapshot>,
32    pub ws_system_connections: Arc<AtomicUsize>,
33    pub snapshots_saved_total: Arc<AtomicU64>,
34    pub shutdown_rx: tokio::sync::oneshot::Receiver<()>,
35}
36
37/// Worker timing and logging config.
38/// Stats logging and pruning use real-time intervals, independent of sample_interval_ms.
39pub struct WorkerConfig {
40    pub sample_interval_ms: u64,
41    /// How often to log app stats (real seconds).
42    pub stats_log_interval_secs: u64,
43    /// How often to prune old data (real seconds).
44    pub prune_interval_secs: u64,
45}
46
47/// Writer config: batching for the dedicated history writer task.
48pub struct HistoryWriterConfig {
49    pub flush_rate: u64,
50    pub flush_interval_secs: u64,
51}

53pub fn spawn(deps: WorkerDeps, config: WorkerConfig) -> tokio::task::JoinHandle<()> {
54    let WorkerDeps {
55        sysinfo_repo,
56        system_info: _,
57        docker_repo,
58        history_repo,
59        tx,
60        write_tx,
61        ws_system_connections,
62        snapshots_saved_total,
63        mut shutdown_rx,
64    } = deps;
65    let WorkerConfig {
66        sample_interval_ms,
67        stats_log_interval_secs,
68        prune_interval_secs,
69    } = config;
70
71    let stats_log_interval = Duration::from_secs(stats_log_interval_secs);
72    let prune_interval = Duration::from_secs(prune_interval_secs);
73
74    tokio::spawn(async move {
75        let mut tick = interval(Duration::from_millis(sample_interval_ms));
76        tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
77        let mut stats_log_tick = interval(stats_log_interval);
78        stats_log_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
79        let mut prune_tick = interval(prune_interval);
80        prune_tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
81
82        let mut snapshots_pruned_total: u64 = 0;
83        let mut last_no_receivers_warn: Option<Instant> = None;
84
85        let worker_span = tracing::span!(tracing::Level::DEBUG, "worker", sample_interval_ms);
86        let _guard = worker_span.enter();
87
88        loop {
89            tokio::select! {
90                _ = tick.tick() => {
91            let timestamp = std::time::SystemTime::now()
92                .duration_since(std::time::UNIX_EPOCH)
93                .map(|d| d.as_millis() as u64)
94                .unwrap_or_else(|e| {
95                    tracing::warn!(
96                        error = %e,
97                        operation = "get_timestamp",
98                        "system time error"
99                    );
100                    0
101                });
102
103            // Degrade gracefully: a single failing collector falls back to defaults for that
104            // metric rather than dropping the whole tick (which would lose the healthy metrics too).
105            let cpu = sysinfo_repo.get_cpu_stats().await.unwrap_or_else(|e| {
106                tracing::warn!(error = %e, operation = "get_cpu_stats", "CPU stats failed; using defaults");
107                Default::default()
108            });
109            let ram = sysinfo_repo.get_ram_stats().await.unwrap_or_else(|e| {
110                tracing::warn!(error = %e, operation = "get_ram_stats", "RAM stats failed; using defaults");
111                Default::default()
112            });
113            let containers = docker_repo.list_running_and_refresh_stats().await;
114            let storage = sysinfo_repo.get_storage_stats().await.unwrap_or_else(|e| {
115                tracing::warn!(error = %e, operation = "get_storage_stats", "storage stats failed; using defaults");
116                Default::default()
117            });
118            let network = sysinfo_repo.get_network_stats().await.unwrap_or_else(|e| {
119                tracing::warn!(error = %e, operation = "get_network_stats", "network stats failed; using defaults");
120                Default::default()
121            });
122            let system = sysinfo_repo.get_system_stats().await.unwrap_or_else(|e| {
123                tracing::warn!(error = %e, operation = "get_system_stats", "system stats failed; using defaults");
124                Default::default()
125            });
126
127            let snapshot = FullSystemSnapshot {
128                timestamp,
129                cpu,
130                ram,
131                containers,
132                storage,
133                network,
134                system,
135            };
136
137            // Only clone for the broadcast when someone is actually listening.
138            if tx.receiver_count() > 0 {
139                let _ = tx.send(snapshot.clone());
140            } else {
141                let should_warn = last_no_receivers_warn
142                    .is_none_or(|t| t.elapsed() >= NO_RECEIVERS_WARN_INTERVAL);
143                if should_warn {
144                    tracing::debug!(
145                        operation = "broadcast_snapshot",
146                        "No active WebSocket clients; skipping broadcast"
147                    );
148                    last_no_receivers_warn = Some(Instant::now());
149                }
150            }
151            if write_tx.send(snapshot).await.is_err() {
152                tracing::debug!("History writer channel closed");
153            }
154                }
155                _ = &mut shutdown_rx => {
156                    tracing::debug!("Worker shutting down");
157                    break;
158                }
159                _ = stats_log_tick.tick() => {
160                    tracing::info!(
161                        ws_system_clients =
162                            ws_system_connections.load(std::sync::atomic::Ordering::Relaxed),
163                        snapshots_saved_total = snapshots_saved_total.load(std::sync::atomic::Ordering::Relaxed),
164                        snapshots_pruned_total = snapshots_pruned_total,
165                        "app stats"
166                    );
167                }
168                _ = prune_tick.tick() => {
169                    if let Err(e) = history_repo.prune_old_data().await {
170                        tracing::warn!(
171                            error = %e,
172                            operation = "prune_old_data",
173                            "Failed to prune old data"
174                        );
175                    } else {
176                        tracing::debug!(operation = "prune_old_data", "Old data pruned successfully");
177                        snapshots_pruned_total += 1;
178                    }
179                }
180            }
181        }
182    })
183}
